

```go
  func ForceFindMaximumSubarray(arr []int, low, high int) (int, int, int){
    if low == high {
      return low, high, arr[low]
    }

    left, right, sum := low, low, arr[low]

    for i := low; i <= high; i++{
      tempSum := arr[i] 

      for j := i + 1; j <= high; j++{
        tempSum += arr[j]

        if tempSum > sum {
          left = i
          right = j
          sum = tempSum
        }
      }
    }
    return left, right, sum
  }
```
